------------------------------------------user----------------------------------------------

CREATE TABLE user_information (
  user_id INT NOT NULL AUTO_INCREMENT,
  full_name VARCHAR(255) NOT NULL,
  date_of_birth DATE NOT NULL,
  gender VARCHAR(255) NOT NULL,
  phone_number VARCHAR(255) NOT NULL,
  email VARCHAR(255) NOT NULL,
  address VARCHAR(255) NOT NULL,
  PRIMARY KEY (user_id)
);

CREATE TABLE educational_backgrounds (
  educational_background_id INT NOT NULL AUTO_INCREMENT,
  user_id INT NOT NULL,
  degree VARCHAR(255) NOT NULL,
  institution VARCHAR(255) NOT NULL,
  PRIMARY KEY (educational_background_id),
  FOREIGN KEY (user_id) REFERENCES user_information(user_id)
);

CREATE TABLE working_experiences (
  working_experience_id INT NOT NULL AUTO_INCREMENT,
  user_id INT NOT NULL,
  job_title VARCHAR(255) NOT NULL,
  employer VARCHAR(255) NOT NULL,
  PRIMARY KEY (working_experience_id),
  FOREIGN KEY (user_id) REFERENCES user_information(user_id)
);

----------------------------------------------job--------------------------------------------------

CREATE TABLE job_description (
  job_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  job_title VARCHAR(255) NOT NULL,
  recruiter VARCHAR(255) NOT NULL,
  application_deadline DATE NOT NULL,
  salary DECIMAL(10,2) NOT NULL,
  working_location VARCHAR(255) NOT NULL
  category VARCHAR(50) NOT NULL,
  applicant_count INT NOT NULL DEFAULT 0  -- Add this column
);

CREATE TABLE job_requirements (
  job_requirement_id INT NOT NULL AUTO_INCREMENT,
  job_id INT NOT NULL,
  requirement TEXT NOT NULL,
  PRIMARY KEY (job_requirement_id),
  FOREIGN KEY (job_id) REFERENCES job_description(job_id)
);

CREATE TABLE job_benefits (
  job_benefit_id INT NOT NULL AUTO_INCREMENT,
  job_id INT NOT NULL,
  benefit TEXT NOT NULL,
  PRIMARY KEY (job_benefit_id),
  FOREIGN KEY (job_id) REFERENCES job_description(job_id)
);

-----------------------------------------------course----------------------------------------------

CREATE TABLE courses (
  course_id INT NOT NULL AUTO_INCREMENT,
  course_name VARCHAR(255) NOT NULL,
  course_length INT NOT NULL,
  course_price DECIMAL(10,2) NOT NULL,
  category VARCHAR(50) NOT NULL,
  enrolment_count INT NOT NULL DEFAULT 0  -- Add this column
  PRIMARY KEY (course_id)
);

CREATE TABLE course_outlines (
  outline_id INT NOT NULL AUTO_INCREMENT,
  course_id INT NOT NULL,
  outline_text TEXT NOT NULL,
  PRIMARY KEY (outline_id),
  FOREIGN KEY (course_id) REFERENCES courses(course_id)
);

CREATE TABLE course_outcomes (
  outcome_id INT NOT NULL AUTO_INCREMENT,
  course_id INT NOT NULL,
  outcome_text TEXT NOT NULL,
  PRIMARY KEY (outcome_id),
  FOREIGN KEY (course_id) REFERENCES courses(course_id)
);
